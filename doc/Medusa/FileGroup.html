def load
  return if @loaded
  struct                  = fetch_body
  @collection_id          = struct['collection_id']
  @directory_id           = struct['cfs_directory']['id'] if struct['cfs_directory']
  @external_file_location = struct['external_file_location']
  @id                     = struct['id']
  @storage_level          = struct['storage_level']
  @title                  = struct['title']
  @uuid                   = struct['uuid']
  @loaded                 = true
end
