def load
  return if @loaded
  struct        = fetch_body
  @id           = struct['id']
  @relative_key = struct['relative_pathname']
  @uuid         = struct['uuid']
  struct['subdirectories'].each do |subdir|
    @directories << Directory.with_id(subdir['id'])
  end
  struct['files'].each do |file|
    @files << File.with_json(file)
  end
  @loaded = true
end
