def load
  return if @loaded
  struct             = fetch_body
  @contact_email     = struct['contact_email']
  @email             = struct['email']
  @home_url          = struct['url']
  @id                = struct['id']
  @ldap_admin_domain = struct['ldap_admin_domain']
  @ldap_admin_group  = struct['ldap_admin_group']
  @title             = struct['title']
  @uuid              = struct['uuid']
  struct['collections'].each do |col_struct|
    @collections << Collection.with_id(col_struct['id'])
  end
  @loaded = true
end
